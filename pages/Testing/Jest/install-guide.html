Jest — Installation Guide

Set up Jest for JavaScript and TypeScript projects. Configure scripts, TypeScript, Babel, coverage, and debug.

NODE TS BABEL COVERAGE
Setup Run TypeScript Babel Coverage Debug

Setup

# Initialize project
npm init -y

# Install Jest	npm i -D jest

# Or yarn/pnpm
yarn add -D jest
pnpm add -D jest

# Optional: create config
npx jest --init

Package Scripts

{
  "scripts": {
    "test": "jest",
    "test:watch": "jest --watch",
    "test:ci": "jest --ci"
  }
}

Run Tests

# Run all tests
npm test

# Watch changed tests
npm run test:watch

# CI mode + JUnit report (example)
jest --ci --reporters=default --reporters=jest-junit

TypeScript

# Option A: ts-jest (no Babel)
npm i -D ts-jest @types/jest typescript
npx ts-jest config:init

# jest.config.ts
export default {
  testEnvironment: 'node',
  preset: 'ts-jest',
}
# Option B: Babel + TS
npm i -D @babel/core @babel/preset-env @babel/preset-typescript babel-jest @types/jest typescript

# babel.config.json
{
  "presets": ["@babel/preset-env", "@babel/preset-typescript"]
}

Babel (JS/TS)

// jest.config.js
module.exports = {
  transform: { '^.+\\.[tj]sx?$': ['babel-jest', { rootMode: 'upward' }] },
  testEnvironment: 'node',
}

Coverage

// jest.config.js
module.exports = {
  collectCoverage: true,
  collectCoverageFrom: ['src/**/*.{js,ts}'],
  coverageReporters: ['text', 'lcov'],
}

Debug in VS Code

{
  "version": "0.2.0",
  "configurations": [
    {
      "type": "node",

      "program": "${workspaceFolder}/node_modules/jest/bin/jest.js",

      "console": "integratedTerminal"

}
