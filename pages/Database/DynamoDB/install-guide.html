AWS Account & CLI

1Create an AWS account if you don’t have one yet.
2Install AWS CLI 
# Windows (winget)
winget install AWS.AWSCLI

	# macOS (Homebrew)
brew install awscli

	# Linux (curl)
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
unzip awscliv2.zip && sudo ./aws/install
3Configure credentials 
aws configure
				# AWS Access Key ID: <YOUR_KEY>
				# AWS Secret Access Key: <YOUR_SECRET>
				# Default region name: us-east-1
				# Default output format: json

Install DynamoDB Local

Option A — Download JAR

# Create a folder and download
mkdir dynamodb-local && cd dynamodb-local
curl -L -o dynamodb_local_latest.zip \
	https://s3.us-west-2.amazonaws.com/dynamodb-local/dynamodb_local_latest.zip
unzip dynamodb_local_latest.zip

	# Run locally (requires Java 8+)
	java -Djava.library.path=./DynamoDBLocal_lib -jar DynamoDBLocal.jar -sharedDb -inMemory -port 8000

Option B — Homebrew (macOS)

brew tap cio-labs/dynamodb-local
brew install dynamodb-local
brew services start dynamodb-local

Run with Docker

# Start container on port 8000
docker run -d --name ddb -p 8000:8000 amazon/dynamodb-local

	# Stop / remove
docker stop ddb && docker rm ddb

Verify & Sample Operations (AWS CLI)

Against Local

# Create table
aws dynamodb create-table \
	--table-name users \
	--attribute-definitions AttributeName=PK,AttributeType=S AttributeName=SK,AttributeType=S \
	--key-schema AttributeName=PK,KeyType=HASH AttributeName=SK,KeyType=RANGE \
	--billing-mode PAY_PER_REQUEST \
	--endpoint-url http://localhost:8000

	# List tables
aws dynamodb list-tables --endpoint-url http://localhost:8000

	# Put item
aws dynamodb put-item \
	--table-name users \
	--item '{"PK":{"S":"USER#42"},"SK":{"S":"PROFILE#2025"},"name":{"S":"Ada"}}' \
	--endpoint-url http://localhost:8000

	# Get item
aws dynamodb get-item \
	--table-name users \
	--key '{"PK":{"S":"USER#42"},"SK":{"S":"PROFILE#2025"}}' \
	--endpoint-url http://localhost:8000

Against AWS

# Create on-demand table (no capacity planning)
aws dynamodb create-table \
	--table-name users \
	--attribute-definitions AttributeName=PK,AttributeType=S AttributeName=SK,AttributeType=S \
	--key-schema AttributeName=PK,KeyType=HASH AttributeName=SK,KeyType=RANGE \
	--billing-mode PAY_PER_REQUEST
