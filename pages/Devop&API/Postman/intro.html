Collections

Environments & Variables

{
  "name": "local",
  "values": [ { "key": "baseUrl", "value": "http://127.0.0.1:3000", "type": "text" } ]
}

Use {{variable}} in URLs, headers, and bodies; set dynamic vars in scripts.

Tests (pm.*)

// Tests tab
pm.test("status is 200", function () {
  pm.response.to.have.status(200);
});

pm.test("has JSON body", () => {
  pm.response.to.be.json;
  const data = pm.response.json();
  pm.expect(data.health).to.eql("ok");
});

Pre-request Scripts

// Pre-request tab
const token = pm.environment.get("token");
pm.request.headers.add({ key: "Authorization", value: `Bearer ${token}` });
pm.environment.set("ts", Date.now());

Newman (CLI)

# Run collection with environment
newman run collection.json -e local.postman_environment.json --reporters cli,junit --reporter-junit-export results.xml
