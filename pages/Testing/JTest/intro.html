Test Anatomy

// src/main/java/com/acme/Calc.java
public class Calc { public int add(int a, int b){ return a + b; } }

// src/test/java/com/acme/CalcTest.java
import org.junit.jupiter.api.*;
import static org.junit.jupiter.api.Assertions.*;

class CalcTest {
  @Test void adds(){ assertEquals(5, new Calc().add(2,3)); }
}

Parameterized Tests

import org.junit.jupiter.params.*;
import org.junit.jupiter.params.provider.ValueSource;

@ParameterizedTest
@ValueSource(strings = {"a","bb","ccc"})
void hasLength(String s){
  Assertions.assertTrue(s.length() > 0);
}

Mockito Basics

import static org.mockito.Mockito.*;

interface Api { String fetch(); }

@Test
void mocks(){
  Api api = mock(Api.class);
  when(api.fetch()).thenReturn("fake");
  Assertions.assertEquals("fake", api.fetch());
  verify(api, times(1)).fetch();
}
